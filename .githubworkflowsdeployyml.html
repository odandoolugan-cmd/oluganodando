name: Deploy with Secret API Key
on:
push:
branches: [main, master]
jobs:
deploy:
runs-on: ubuntu-latest
permissions:
contents: write
pages: write
id-token: write
steps:
- name: Checkout repository
uses: actions/checkout@v4
- name: Inject API Key into dinastia.html
run: |
sed -i "s|__GROQ_API_KEY__|${{ secrets.GROQ_API_KEY }}|g"
dinastia.html
- name: List files (debug)
run: |
ls -la
echo "=== Verificando que la key se inyectó ==="
grep -n "GROQ_API_KEY" dinastia.html || echo "Key inyectada
correctamente"
- name: Upload artifact
uses: actions/upload-pages-artifact@v3
with:
path: ./
- name: Deploy to GitHub Pages
uses: actions/deploy-pages@v4
